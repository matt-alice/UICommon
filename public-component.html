You can reuse the above UI widgets very easily with our custom .displayMsg() function. 

    < link  rel="stylesheet" type="text/css" href="css/ui-public.css" / >
    < script src="//ajax.googleapis.com/ajax/libs/jquery/1.11.1/jquery.min.js" >< /script >
    < script type="text/javascript"  src="js/ui-public.js" >< /script >
    
    < script >
        $('#message-success').displayMsg({
            msgType: 'success',
            msgTitle: 'Congratulations!'
        });

        $('#message-warning').displayMsg({
            msgType: 'warning',
            msgTitle: 'Heads up!!'
        });

        $('#message-error').displayMsg({
            msgType: 'error',
            msgTitle: 'Error!'
        });
    < /script >
class "ajax-response loading"
